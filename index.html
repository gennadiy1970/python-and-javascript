Learning by comparing Python and JavaScript

Both languages are very popular. And if you know one of them, then you can easily understand code examples from another language.

The goal of this article is to help you quickly master the technique of a programming language you don't know yet, comparing it with similar code in a language you understand.

If

Python


leftOperand = 6
rightOperand = 2
operator = '+'
result = ''

if operator == '+':
    result = str(leftOperand + rightOperand)
elif operator == '-':     
    result = str(leftOperand - rightOperand)
else: result = 'Неверный оператор'

print(result)

Javascript


const leftOperand = 6;
const rightOperand = 2;
const operator = '+';
let result = '';

if (operator === '+') {
  result = leftOperand + rightOperand;
} else if (operator === '-') {
  result = leftOperand - rightOperand;
} else {
  result = 'Неверный оператор';
}
console.log(result);

Lambda function and filter method

Python


# primitives                
print(list(
  filter(
    lambda x: x%2, 
    [10, 111, 102, 213, 314, 515]
  )
))
# [111, 213, 515]

# objects
elements = [
    { 
      'atom number': 1, 
      'name': 'Hydrogen', 
    },
    { 
      'atom number': 2, 
      'name': 'Helium', 
    }
]

print(list(
    filter(
        lambda x: x['atom number'] == 2, 
        elements
    )
))
##[
##  {
##    'atom number': 2, 
##    'name': 'Helium',
##  }
##]

Javascript


// primitives                
console.log(
  [10, 111, 102, 213, 314, 515]
  .filter(
    x => x % 2 !== 0
  )
)
// [111, 213, 515]

// objects
elements = [
    { 
      'atom number': 1, 
      'name': 'Hydrogen', 
    },
    { 
      'atom number': 2, 
      'name': 'Helium', 
    }
]

console.log(
  elements
  .filter(
    x => x['atom number'] === 2
  )
)
/*[
    {
      'atom number': 2, 
      'name': 'Helium',
    }
]*/
